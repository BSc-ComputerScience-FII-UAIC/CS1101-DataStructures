//comparati 2 liste 
struct nod {
		int info;
		nod * urm;
	};

struct lista {
		nod * prim;
		nod * ultim;
		int lung;
};

bool liste_identice(lista L1, lista L2)
{
n=L1.lung;
if (L1.lung != L2.lung)
    return false;
else
{
  nod* p1=L1.prim;
  nod* p2=L2.prim;
  for (i=1;i<=n;i++)
  {
      if (L1.info != L2. info)
           return false;

      p1 = p1.urm;
      p2 = p2.urm;
  }
}
return true;
}
