//elementul de sub diagonala secundara dintr o matrice
matrice(a,n)
{
min=a[0][0];
for (i=0;i<n;i++)
{
for (j=0;j<n;j++)
{
if (min > a[i][n-i+1])

      min=a[i][n-i+1];
}
}
return min;
}
